Matrix.css: The Ultimate Cyberpunk CSS Framework

                    
Matrix.css is a CSS framework that brings the look and feel of The Matrix and late-90s cyberpunk hacker culture to modern web design. Transform your website into a fully immersive digital experience with neon-green-on-black terminal aesthetics while maintaining clean, semantic HTML5, responsive layouts, and accessible UX.

                    
                    

                        Note: Matrix.css is designed to work with modern browsers and provides both dark (default) and light themes.
                    

                    
                    
Key Features

                    

                        

                            

                                

                                    
Matrix Aesthetic

                                    
Authentic Matrix-inspired design with neon green on black, terminal styling, and cyberpunk elements.

                                

                            

                        

                        

                            

                                

                                    
Modern Framework

                                    
Built with responsive design, accessibility, and performance in mind, compatible with modern browsers.

                                

                            

                        

                        

                            

                                

                                    
Special Effects

                                    
CRT scanlines, screen flicker, glitch text, and other special effects to enhance the retro experience.

                                

                            

                        

                    

                


                
                

                    
Installation

                    
There are several ways to include Matrix.css in your project:

                    
                    
1. Direct Download

                    
Download the CSS file and include it in your HTML:

                    
Copy<link rel="stylesheet" href="path/to/matrix.css">

                    
                    
2. CDN

                    
Include directly from a CDN:

                    
Copy<link rel="stylesheet" href="https://cdn.example.com/matrix-css/v2.0/matrix.css">

                    
                    
3. NPM

                    
Install via npm:

                    
Copynpm install matrix-css

                    
                    
Then import it in your project:

                    
Copyimport 'matrix-css/dist/matrix.css';

<script>
  // JavaScript for handling terminal input
  document.getElementById('terminal-input').addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
      const cmd = this.value;
      const output = document.getElementById('terminal-output');
      
      // Add the command to the output
      output.innerHTML += `\n$ ${cmd}`;
      
      // Process the command
      if (cmd === 'help') {
        output.innerHTML += '\nAvailable commands: help, version, clear';
      } else if (cmd === 'version') {
        output.innerHTML += '\nMatrix.css v2.0';
      } else if (cmd === 'clear') {
        output.innerHTML = '';
      } else if (cmd.trim() !== '') {
        output.innerHTML += `\nCommand not found: ${cmd}`;
      }
      
      // Clear the input
      this.value = '';
      
      // Scroll to bottom
      output.scrollTop = output.scrollHeight;
    }
  });
</script>

Neural Network

                    
Create a visual representation of a neural network with activating nodes and connections.

                    
                    

                        

                            
Neural Network Visualizer

                        

                        

                            

                                

                                
                            

                        

                        

                            
The neural network visualizer combines HTML structure with JavaScript to create an animated representation of a neural network.

                        

                        
Copy<div class="neural-network-container" id="neural-network">
  <div class="neural-network-bg"></div>
  <!-- Nodes and connections will be added by JavaScript -->
</div>

<script>
  // Neural network visualization code
  function initNeuralNetwork() {
    const container = document.getElementById('neural-network');
    const layers = [3, 4, 4, 2]; // Network architecture
    
    // Create nodes for each layer
    layers.forEach((nodeCount, layerIndex) => {
      const layerPosition = (layerIndex + 1) / (layers.length + 1);
      
      for (let i = 0; i < nodeCount; i++) {
        const verticalPosition = (i + 1) / (nodeCount + 1);
        
        // Create node
        const node = document.createElement('div');
        node.className = 'neural-network-node';
        node.style.left = `${layerPosition * 100}%`;
        node.style.top = `${verticalPosition * 100}%`;
        node.dataset.layer = layerIndex;
        node.dataset.index = i;
        
        container.appendChild(node);
        
        // Create connections to previous layer
        if (layerIndex > 0) {
          const prevLayer = layerIndex - 1;
          const prevNodeCount = layers[prevLayer];
          
          for (let j = 0; j < prevNodeCount; j++) {
            // Calculate positions
            const startX = (prevLayer + 1) / (layers.length + 1) * 100;
            const startY = (j + 1) / (prevNodeCount + 1) * 100;
            const endX = layerPosition * 100;
            const endY = verticalPosition * 100;
            
            // Create connection
            const connection = document.createElement('div');
            connection.className = 'neural-network-connection';
            
            // Calculate dimensions and position
            const length = Math.sqrt(Math.pow(endX - startX, 2) + Math.pow(endY - startY, 2));
            const angle = Math.atan2(endY - startY, endX - startX) * 180 / Math.PI;
            
            connection.style.width = `${length}%`;
            connection.style.left = `${startX}%`;
            connection.style.top = `${startY}%`;
            connection.style.transform = `rotate(${angle}deg)`;
            
            container.appendChild(connection);
          }
        }
      }
    });
    
    // Animate the network
    function animateNetwork() {
      // Randomly activate nodes
      const nodes = container.querySelectorAll('.neural-network-node');
      const connections = container.querySelectorAll('.neural-network-connection');
      
      // Randomly activate an input node
      const inputNodes = Array.from(nodes).filter(node => node.dataset.layer === '0');
      const randomNode = inputNodes[Math.floor(Math.random() * inputNodes.length)];
      
      randomNode.classList.add('active');
      
      // Propagate the activation through the network
      setTimeout(() => {
        propagateActivation(randomNode, nodes, connections);
      }, 500);
    }
    
    function propagateActivation(node, nodes, connections) {
      const layer = parseInt(node.dataset.layer);
      const index = parseInt(node.dataset.index);
      
      // Find connections from this node to the next layer
      const outgoingConnections = Array.from(connections).filter(conn => {
        const start = conn.style.left.replace('%', '') / 100 * container.clientWidth;
        const top = conn.style.top.replace('%', '') / 100 * container.clientHeight;
        
        const nodeLeft = node.style.left.replace('%', '') / 100 * container.clientWidth;
        const nodeTop = node.style.top.replace('%', '') / 100 * container.clientHeight;
        
        return Math.abs(start - nodeLeft) < 5 && Math.abs(top - nodeTop) < 5;
      });
      
      // Activate those connections
      outgoingConnections.forEach(conn => {
        conn.classList.add('active');
        
        // Calculate endpoint of connection
        const length = parseFloat(conn.style.width);
        const angle = parseFloat(conn.style.transform.replace('rotate(', '').replace('deg)', '')) * Math.PI / 180;
        const startX = parseFloat(conn.style.left) / 100 * container.clientWidth;
        const startY = parseFloat(conn.style.top) / 100 * container.clientHeight;
        
        const endX = startX + length / 100 * container.clientWidth * Math.cos(angle);
        const endY = startY + length / 100 * container.clientWidth * Math.sin(angle);
        
        // Find the node at this endpoint
        const nextNode = Array.from(nodes).find(n => {
          const nLeft = parseFloat(n.style.left) / 100 * container.clientWidth;
          const nTop = parseFloat(n.style.top) / 100 * container.clientHeight;
          
          return Math.abs(nLeft - endX) < 15 && Math.abs(nTop - endY) < 15;
        });
        
        if (nextNode) {
          // Activate the next node after a delay
          setTimeout(() => {
            nextNode.classList.add('active');
            propagateActivation(nextNode, nodes, connections);
            
            // Reset this connection after a delay
            setTimeout(() => {
              conn.classList.remove('active');
            }, 500);
          }, 200);
        }
      });
      
      // Reset this node after a delay
      setTimeout(() => {
        node.classList.remove('active');
      }, 1000);
    }
    
    // Start animation
    setInterval(animateNetwork, 3000);
  }
  
  // Initialize the neural network on page load
  document.addEventListener('DOMContentLoaded', initNeuralNetwork);
</script>

  // JavaScript for handling terminal input
  document.getElementById('terminal-input').addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
      const cmd = this.value;
      const output = document.getElementById('terminal-output');
      
      // Add the command to the output
      output.innerHTML += `\n$ ${cmd}`;
      
      // Process the command
      if (cmd === 'help') {
        output.innerHTML += '\nAvailable commands: help, version, clear';
      } else if (cmd === 'version') {
        output.innerHTML += '\nMatrix.css v2.0';
      } else if (cmd === 'clear') {
        output.innerHTML = '';
      } else if (cmd.trim() !== '') {
        output.innerHTML += `\nCommand not found: ${cmd}`;
      }
      
      // Clear the input
      this.value = '';
      
      // Scroll to bottom
      output.scrollTop = output.scrollHeight;
    }
  });

Neural Network Visualizer

                        

                        

                            

                                

                                
                            

                        

                        

                            
The neural network visualizer combines HTML structure with JavaScript to create an animated representation of a neural network.

                        

                        
Copy<div class="neural-network-container" id="neural-network">
  <div class="neural-network-bg"></div>
  <!-- Nodes and connections will be added by JavaScript -->

<script>
  // Neural network visualization code
  function initNeuralNetwork() {
    const container = document.getElementById('neural-network');
    const layers = [3, 4, 4, 2]; // Network architecture
    
    // Create nodes for each layer
    layers.forEach((nodeCount, layerIndex) => {
      const layerPosition = (layerIndex + 1) / (layers.length + 1);
      
      for (let i = 0; i < nodeCount; i++) {
        const verticalPosition = (i + 1) / (nodeCount + 1);
        
        // Create node
        const node = document.createElement('div');
        node.className = 'neural-network-node';
        node.style.left = `${layerPosition * 100}%`;
        node.style.top = `${verticalPosition * 100}%`;
        node.dataset.layer = layerIndex;
        node.dataset.index = i;
        
        container.appendChild(node);
        
        // Create connections to previous layer
        if (layerIndex > 0) {
          const prevLayer = layerIndex - 1;
          const prevNodeCount = layers[prevLayer];
          
          for (let j = 0; j < prevNodeCount; j++) {
            // Calculate positions
            const startX = (prevLayer + 1) / (layers.length + 1) * 100;
            const startY = (j + 1) / (prevNodeCount + 1) * 100;
            const endX = layerPosition * 100;
            const endY = verticalPosition * 100;
            
            // Create connection
            const connection = document.createElement('div');
            connection.className = 'neural-network-connection';
            
            // Calculate dimensions and position
            const length = Math.sqrt(Math.pow(endX - startX, 2) + Math.pow(endY - startY, 2));
            const angle = Math.atan2(endY - startY, endX - startX) * 180 / Math.PI;
            
            connection.style.width = `${length}%`;
            connection.style.left = `${startX}%`;
            connection.style.top = `${startY}%`;
            connection.style.transform = `rotate(${angle}deg)`;
            
            container.appendChild(connection);
          }
        }
      }
    });
    
    // Animate the network
    function animateNetwork() {
      // Randomly activate nodes
      const nodes = container.querySelectorAll('.neural-network-node');
      const connections = container.querySelectorAll('.neural-network-connection');
      
      // Randomly activate an input node
      const inputNodes = Array.from(nodes).filter(node => node.dataset.layer === '0');
      const randomNode = inputNodes[Math.floor(Math.random() * inputNodes.length)];
      
      randomNode.classList.add('active');
      
      // Propagate the activation through the network
      setTimeout(() => {
        propagateActivation(randomNode, nodes, connections);
      }, 500);
    }
    
    function propagateActivation(node, nodes, connections) {
      const layer = parseInt(node.dataset.layer);
      const index = parseInt(node.dataset.index);
      
      // Find connections from this node to the next layer
      const outgoingConnections = Array.from(connections).filter(conn => {
        const start = conn.style.left.replace('%', '') / 100 * container.clientWidth;
        const top = conn.style.top.replace('%', '') / 100 * container.clientHeight;
        
        const nodeLeft = node.style.left.replace('%', '') / 100 * container.clientWidth;
        const nodeTop = node.style.top.replace('%', '') / 100 * container.clientHeight;
        
        return Math.abs(start - nodeLeft) < 5 && Math.abs(top - nodeTop) < 5;
      });
      
      // Activate those connections
      outgoingConnections.forEach(conn => {
        conn.classList.add('active');
        
        // Calculate endpoint of connection
        const length = parseFloat(conn.style.width);
        const angle = parseFloat(conn.style.transform.replace('rotate(', '').replace('deg)', '')) * Math.PI / 180;
        const startX = parseFloat(conn.style.left) / 100 * container.clientWidth;
        const startY = parseFloat(conn.style.top) / 100 * container.clientHeight;
        
        const endX = startX + length / 100 * container.clientWidth * Math.cos(angle);
        const endY = startY + length / 100 * container.clientWidth * Math.sin(angle);
        
        // Find the node at this endpoint
        const nextNode = Array.from(nodes).find(n => {
          const nLeft = parseFloat(n.style.left) / 100 * container.clientWidth;
          const nTop = parseFloat(n.style.top) / 100 * container.clientHeight;
          
          return Math.abs(nLeft - endX) < 15 && Math.abs(nTop - endY) < 15;
        });
        
        if (nextNode) {
          // Activate the next node after a delay
          setTimeout(() => {
            nextNode.classList.add('active');
            propagateActivation(nextNode, nodes, connections);
            
            // Reset this connection after a delay
            setTimeout(() => {
              conn.classList.remove('active');
            }, 500);
          }, 200);
        }
      });
      
      // Reset this node after a delay
      setTimeout(() => {
        node.classList.remove('active');
      }, 1000);
    }
    
    // Start animation
    setInterval(animateNetwork, 3000);
  }
  
  // Initialize the neural network on page load
  document.addEventListener('DOMContentLoaded', initNeuralNetwork);
</script>
